MineRLTreechop-v0¶

Usage¶

import gym
import minerl

env = gym.make("MineRLTreechop-v0") # A MineRLTreechop-v0 env

obs, _ = env.reset()
done = False

while not done:
    # Take a no-op through the environment.
    env.step(env.action_space.noop)
